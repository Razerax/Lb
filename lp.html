                return false;

                return false;

            let n = document.getElementById("numberInput").value;
            n = parseInt(n);

            if (isNaN(n) || n <= 0) {
                alert("Please enter a valid positive number.");
                return;
            }

            let sum = 0;
            for (let i = 1; i <= n; i++) {

9. Prime & Fibo
<?php
function generatePrimes($maxNumber) {
    echo "Prime numbers up to " . $maxNumber . ": ";
    $primes = array();
    
    for ($number = 2; $number <= $maxNumber; $number++) {
        $isPrime = true;

        for ($i = 2; $i * $i <= $number; $i++) {
            if (($number % $i) == 0) {
                $isPrime = false;
                break;
            }
        }
        
        if ($isPrime) {
            $primes[] = $number;
        }
    }
    
    echo implode(' ', $primes);
    echo "<br>";
}

function FibonacciSeries($numTerms) {
    echo "Fibonacci Series (" . $numTerms . " terms): ";
    
    $first = 0;
    $second = 1;
    
    if ($numTerms >= 1) {
        echo $first . " ";
    }
    if ($numTerms >= 2) {
        echo $second . " ";
    }
    
    for ($i = 3; $i <= $numTerms; $i++) {
        $next = $first + $second;
        echo $next . " ";
        
        $first = $second;
        $second = $next;
    }
    echo "<br>";
}

generatePrimes(50);
FibonacciSeries(10);
?>

12. Search Criteria
<?php
$data = [
    [ 'id' => 1, 'name' => 'Srikantn', 'age' => 30 ],
    [ 'id' => 2, 'name' => 'Vikrant', 'age' => 28 ],
    [ 'id' => 3, 'name' => 'Srinivas', 'age' => 50 ],
    [ 'id' => 4, 'name' => 'Sanjay', 'age' => 45 ],
    [ 'id' => 5, 'name' => 'Saathvik', 'age' => 18 ]
];

function searchByCriteria(array $data, $criteria) {
    $results = [];

    foreach ($data as $entry) {
        $match = true;

        foreach ($criteria as $key => $value) {
            if (isset($entry[$key])) {
                if ($key === 'age' && strpos($value, '>=') === 0) {
                    $minAge = (int)substr($value, 2);
                    if ($entry['age'] < $minAge) {
                        $match = false;
                        break;
                    }
                } elseif ($entry[$key] != $value) {
                    $match = false;
                    break;
                }
            } else {
                $match = false;
                break;
            }
        }

        if ($match) {
            $results[] = $entry;
        }
    }
    return $results;
}

$criteria1 = ['age' => '>= 50'];
$results1 = searchByCriteria($data, $criteria1);

$criteria2 = ['name' => 'Sanjay'];
$results2 = searchByCriteria($data, $criteria2);

echo "Results 1 (Age >= 50): ";
print_r($results1);
echo "<br>";

echo "Results 2 (Name = Sanjay): ";
print_r($results2);
?>

function generateCaptchaCode($length) {
    $characters = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ';
    $characterStringLength = strlen($characters);
    $randomString = '';

    for ($i = 0; $i < $length; $i++) {
        $randomString .= $characters[rand(0, $characterStringLength - 1)];
    }

    return $randomString;
}
