9. Prime & Fibo
<?php
function generatePrimes($maxNumber) {
    echo "Prime numbers up to " . $maxNumber . ": ";
    $primes = array();
    
    for ($number = 2; $number <= $maxNumber; $number++) {
        $isPrime = true;

        for ($i = 2; $i * $i <= $number; $i++) {
            if (($number % $i) == 0) {
                $isPrime = false;
                break;
            }
        }
        
        if ($isPrime) {
            $primes[] = $number;
        }
    }
    
    echo implode(' ', $primes);
    echo "<br>";
}

function FibonacciSeries($numTerms) {
    echo "Fibonacci Series (" . $numTerms . " terms): ";
    
    $first = 0;
    $second = 1;
    
    if ($numTerms >= 1) {
        echo $first . " ";
    }
    if ($numTerms >= 2) {
        echo $second . " ";
    }
    
    for ($i = 3; $i <= $numTerms; $i++) {
        $next = $first + $second;
        echo $next . " ";
        
        $first = $second;
        $second = $next;
    }
    echo "<br>";
}

generatePrimes(50);
FibonacciSeries(10);
?>

12. Search Criteria
<?php
$data = [
    [ 'id' => 1, 'name' => 'Srikantn', 'age' => 30 ],
    [ 'id' => 2, 'name' => 'Vikrant', 'age' => 28 ],
    [ 'id' => 3, 'name' => 'Srinivas', 'age' => 50 ],
    [ 'id' => 4, 'name' => 'Sanjay', 'age' => 45 ],
    [ 'id' => 5, 'name' => 'Saathvik', 'age' => 18 ]
];

function searchByCriteria(array $data, $criteria) {
    $results = [];

    foreach ($data as $entry) {
        $match = true;

        foreach ($criteria as $key => $value) {
            if (isset($entry[$key])) {
                if ($key === 'age' && strpos($value, '>=') === 0) {
                    $minAge = (int)substr($value, 2);
                    if ($entry['age'] < $minAge) {
                        $match = false;
                        break;
                    }
                } elseif ($entry[$key] != $value) {
                    $match = false;
                    break;
                }
            } else {
                $match = false;
                break;
            }
        }

        if ($match) {
            $results[] = $entry;
        }
    }
    return $results;
}

$criteria1 = ['age' => '>= 50'];
$results1 = searchByCriteria($data, $criteria1);

$criteria2 = ['name' => 'Sanjay'];
$results2 = searchByCriteria($data, $criteria2);

echo "Results 1 (Age >= 50): ";
print_r($results1);
echo "<br>";

echo "Results 2 (Name = Sanjay): ";
print_r($results2);
?>

function generateCaptchaCode($length) {
    $characters = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ';

<!DOCTYPE html>
<html>
<head>
<title>PHP Exception Handling</title>
<style>
body { font-family: Arial; background:#f4f4f4; padding:30px; }

 background:#fff; padding:20px; border:1px solid #ccc;
 box-shadow:2px 2px 10px #aaa; max-width:500px; margin:auto;
}
</style>

<?php
$num1 = $_POST['num1'] ?? '';
$num2 = $_POST['num2'] ?? '';
$date_input = $_POST['date_input'] ?? '';
?>

<div class="form-box">
<h2>Exception Handling Form</h2>
<form method="post">
<label>Enter numerator:</label>
<input type="text" name="num1" value="<?php echo $num1; ?>" required>

<label>Enter denominator:</label>
<input type="text" name="num2" value="<?php echo $num2; ?>" required>

<label>Enter date (YYYY-MM-DD):</label>
<input type="text" name="date_input" value="<?php echo $date_input; ?>" required>

<input type="submit" name="submit" value="Submit">
</form>
</div>

<?php
if (isset($_POST['submit'])) {
    echo '<div class="result-box">';

    // Division handling
    try {
        if (!is_numeric($num1) || !is_numeric($num2)) {
            throw new Exception("Division error: Inputs must be numbers.");
        }
        if ($num2 == 0) {
            throw new Exception("Division error: Cannot divide by zero.");
        }
        echo "Division result: $num1 / $num2 = " . ($num1 / $num2) . "<br>";
    } catch (Exception $e) {
        echo $e->getMessage() . "<br>";
    }

    // Date validation
    try {
        $format = 'Y-m-d';
        $d = DateTime::createFromFormat($format, $date_input);
        if ($d && $d->format($format) === $date_input) {
            echo "The date $date_input is valid.<br>";
        } else {
            throw new Exception("Date error: Invalid date format.");
        }
    } catch (Exception $e) {
        echo $e->getMessage() . "<br>";
    }

    echo '</div>';
}
?>

</body>
</html>
